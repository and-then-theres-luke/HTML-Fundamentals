Here are some notes for HTML

Input types:

    Text inputs
        <input type="text" name="first_name">
    Number inputs  
        <input type="number" name="age">
    Passwords
        <input type="password" name="password">
    Date *not supported on all browsers*
        <input type="date" name="dob">
    Color
        <input type="color" name="text_color">
    Radio
        Bold
            <input type="radio" name="font" value="bold" id="bold">
            <label for="bold">Bold</label>
        Normal
            <input type="radio" name="font" value="normal" id="normal">
            <label for="normal">Normal</label>
    Textarea
        <textarea name="comment" cols="20" rows="3">
        </textarea>
    Select
        <select name="snack">
            <option value="Almonds">Almonds</option>
            <option value="Cheese and Crackers">Cheese and Crackers</option>
            <option value="Pita and Hummus">Pita and Hummus</option>
            <option value="Pears">Pears</option>
        </select>
